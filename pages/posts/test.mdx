```tsx {1-3,4}
import * as React from 'react';
import { Button as UDButton } from '@universe-design/react';
import cn from 'classnames';
import * as ICard from '../../interface';
import { prefix } from '../../constants';
import { openLink } from '../../utils';
import './index.less';

interface ButtonProps extends Omit<ICard.ElementButton, 'text'> {
  onClick?: (e: React.MouseEvent) => void;
  stopPropagation?: 'true'|'false';
}
export const Button: React.FC<ButtonProps> = (props) => {
  return (
    <UDButton
      onClick={(e: React.MouseEvent) => {
        if(props.url) {
          openLink(props.url);
        }
        if(props.stopPropagation === 'true') {
          e.stopPropagation();
        }
        if(props.onClick) {
          props.onClick(e);
        }
      }}
      {...props}
      className={cn(`${prefix}-element-button`, props.className)}
    >
      {props.children}
    </UDButton>
  );
};
```
